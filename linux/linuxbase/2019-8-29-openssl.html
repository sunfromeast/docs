# 使用openssl生成自签名证书

# 1 生成根秘钥
$ openssl genrsa -out ca.key 2048

# 2 创建证书签名请求（csr）
$ openssl req -new -key ca.key -out ca.csr -sub='/CN=192.168.144.3'

# 3 签署自己的证书
$ oepnssl x509 -req -days 3650 -in ca.csr -signkey ca.key -out ca.crt
